echo "id,client_before_create_ms,initial_time_ms,true_startup_time_ms" > startup_log.csv

for i in $(seq 1 30); do
  kubectl delete pod startup-test --ignore-not-found=true

  # client-side timestamp (before creation)
  client_before=$(date +%s%3N)

  kubectl apply -f test_pod.yaml

  echo "Run $i"

  kubectl wait --for=condition=Ready pod/startup-test --timeout=120s

  pod_start=$(kubectl get pod startup-test -o jsonpath='{.status.startTime}')
  initial_ms=$(date -d "$pod_start" +%s%3N)

  true_ms=$(kubectl logs startup-test | head -n 1)

  echo "$i,$client_before,$initial_ms,$true_ms" >> startup_log.csv

  sleep 5
done

test_pod.yaml
####
apiVersion: v1
kind: Pod
metadata:
  name: startup-test
spec:
  containers:
  - name: test
    image: python:3.11-slim
    command:
      - python
      - -u
      - -c
      - |
        import time
        print(int(time.time_ns() / 1_000_000))
        print("STARTED")
        import time
        time.sleep(3600)
  restartPolicy: Never

import pandas as pd
import matplotlib.pyplot as plt

# load data
df = pd.read_csv("startup_log.csv")

# ensure numeric
for c in ["client_before_create_ms", "initial_time_ms", "true_startup_time_ms"]:
    df[c] = pd.to_numeric(df[c], errors="coerce")

# compute deltas
df["client_delta_ms"] = (df["true_startup_time_ms"] - df["client_before_create_ms"])/ 10000
df["k8s_delta_ms"] = (df["true_startup_time_ms"] -  df["initial_time_ms"]) / 1000

# averages
avg_client = df["client_delta_ms"].mean()
avg_k8s = df["k8s_delta_ms"].mean()

print("Avg client_before - true:", avg_client)
print("Avg initial_time - true:", avg_k8s)

plt.figure()

plt.plot(df["id"], df["client_delta_ms"], label="client_before - true (s)")
plt.plot(df["id"], df["k8s_delta_ms"], label="initial_time - true (s)")

plt.axhline(avg_client)
plt.axhline(avg_k8s)

plt.xlabel("Run id")
plt.ylabel("Latency (seconds)")
plt.legend()
plt.tight_layout()

# save file
plt.savefig("startup_latency.png", dpi=300, bbox_inches="tight")

plt.show()
